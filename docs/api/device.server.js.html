YUI.add('mojito-device-addon', function(Y, NAME) {

    /**
     * <strong>Access point:</strong> <em>ac.device.get()</em>
     * Device catalog addon
     * @class Device.common
     */
    function DeviceAddon(command, adapter, ac) {

        var req,
        self = this;

        this._ac = ac;
        this.command = command;

        req = ac.http.getRequest();

        // Set device accessor
        Object.defineProperty(ac, "device",
        {
            get : function() {
                if (req.device) {
                    return req.device;
                } else {
                    return self;
                }
            },
            set : function() {
            },
            configurable: true
        });

    }

    DeviceAddon.prototype = {

        // Intentionally commented out in order to make it instantiable on demand
//        namespace: 'device',

        /**
         * Returns the attribute of the catalog for the device
         * this request was intiated from.
         *
         * @method get
         * @param attribute {String} - The name of the catalog attribute e.g "make" or "model".
         * @returns value
         */
        get : function(attribute) {
            //TODO: make an RPC call.
        }
    };

    DeviceAddon.dependsOn = ['config', 'http'];

    Y.mojito.addons.ac.device = DeviceAddon;

}, '0.1.0', {requires: ['mojito']});
