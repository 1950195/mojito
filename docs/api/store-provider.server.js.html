YUI.add('dali-store-provider-addon', function(Y, NAME) {

    function Addon(command, adapter, ac) {
        this.instance = command.instance;
        this.scripts = {};
        this.ac = ac;
        this.dispatch = ac.dispatch;
    }

    Addon.prototype = {

        namespace: 'store',

        /**
         * Declaration of store requirement.
         * @private
         * @param rs {ResourceStore}
         */
        setStore: function(rs) {
            this.rs = rs;
            if (rs) {
                Y.log('Initialized and activated with Resource Store', 'info', NAME);
            }
        },

        getStore: function() {
            return this.rs;
        }
    };

    Y.mojito.addons.ac.store = Addon;

}, '0.1.0');
